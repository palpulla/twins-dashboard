  <script>
    (() => {
      const form = document.querySelector('#application-form');
      const steps = [...form.querySelectorAll('.form-step')];
      const bars = [...form.querySelectorAll('.progress-step')];
      const progressLabel = document.querySelector('#progress-label');
      const successPanel = document.querySelector('#success-panel');
      const formAlert = document.querySelector('#form-alert');
      const submitButton = document.querySelector('#submit-button');
      let currentStep = 0;

      const labels = {
        'service-repair': 'Service and repair',
        'installation': 'Installations',
        'sales-estimates': 'Sales and estimates',
        'customer-care-operations': 'Customer care and operations',
        'other': 'Something else',
        'madison-area': 'Madison area',
        'milwaukee-area': 'Milwaukee area',
        'open-to-either': 'Open to either',
        'none': 'New to this kind of work',
        'under-1': 'Less than 1 year',
        '1-3': '1 to 3 years',
        '4-7': '4 to 7 years',
        '8-plus': '8 or more years',
        'full-time': 'Full-time',
        'part-time': 'Part-time',
        'either': 'Open to either',
        'text': 'Text message',
        'email': 'Email',
        'phone': 'Phone call',
        'WI': 'Wisconsin',
        'IL': 'Illinois',
        'KY': 'Kentucky',
        'OTHER': 'Other'
      };

      const showStep = (index) => {
        currentStep = index;
        steps.forEach((step, i) => { step.hidden = i !== index; });
        bars.forEach((bar, i) => bar.classList.toggle('active', i <= index));
        progressLabel.textContent = `Step ${index + 1} of ${steps.length}`;
        formAlert.classList.remove('show');
        document.querySelector('.form-card').scrollIntoView({ behavior: 'smooth', block: 'start' });
        const heading = steps[index].querySelector('h3');
        if (heading) { heading.setAttribute('tabindex', '-1'); heading.focus({ preventScroll: true }); }
      };

      const setError = (input, message) => {
        input.setAttribute('aria-invalid', 'true');
        const field = input.closest('.field');
        const target = field?.querySelector('.error-text');
        if (target) target.textContent = message;
      };

      const clearErrors = (step) => {
        step.querySelectorAll('[aria-invalid="true"]').forEach((input) => input.removeAttribute('aria-invalid'));
        step.querySelectorAll('.error-text').forEach((error) => { error.textContent = ''; });
      };

      const validateStep = (step) => {
        clearErrors(step);
        let valid = true;
        let firstInvalid = null;

        step.querySelectorAll('[data-required-group]').forEach((group) => {
          const name = group.dataset.requiredGroup;
          const checked = group.querySelector(`input[name="${name}"]:checked`);
          if (!checked) {
            valid = false;
            firstInvalid ||= group.querySelector('input');
            const error = step.querySelector(`[data-error-for="${name}"]`);
            if (error) error.textContent = 'Please choose one option.';
          }
        });

        step.querySelectorAll('input[required]:not([type="radio"]):not([type="checkbox"]), select[required], textarea[required]').forEach((input) => {
          if (!input.checkValidity()) {
            valid = false;
            firstInvalid ||= input;
            const message = input.validity.typeMismatch ? 'Please enter a valid value.' : 'This field is required.';
            setError(input, message);
          }
        });

        if (!valid && firstInvalid) firstInvalid.focus();
        return valid;
      };

      const valueOf = (name) => new FormData(form).get(name) || '';
      const display = (value) => labels[value] || value || 'Not provided';

      const renderReview = () => {
        const rows = [
          ['Area of interest', display(valueOf('roleInterest'))],
          ['Preferred market', display(valueOf('market'))],
          ['Name', `${valueOf('firstName')} ${valueOf('lastName')}`.trim()],
          ['Email', valueOf('email')],
          ['Mobile phone', valueOf('phone')],
          ['Location', `${valueOf('city')}, ${display(valueOf('state'))}`],
          ['Preferred contact', display(valueOf('preferredContact'))],
          ['Related experience', display(valueOf('experience'))],
          ['Work preference', display(valueOf('workType'))],
          ['Available to start', valueOf('availableDate') || 'Not provided']
        ];
        const review = document.querySelector('#review');
        review.innerHTML = rows.map(([term, detail]) => `<div class="review-row"><dt>${term}</dt><dd>${detail}</dd></div>`).join('');
      };

      form.querySelectorAll('.next-button').forEach((button) => {
        button.addEventListener('click', () => {
          if (!validateStep(steps[currentStep])) return;
          if (currentStep === 2) renderReview();
          showStep(Math.min(currentStep + 1, steps.length - 1));
        });
      });

      form.querySelectorAll('.back-button').forEach((button) => {
        button.addEventListener('click', () => showStep(Math.max(currentStep - 1, 0)));
      });

      const params = new URLSearchParams(location.search);
      document.querySelector('#sourceUrl').value = location.href;
      document.querySelector('#utmSource').value = params.get('utm_source') || '';
      document.querySelector('#utmCampaign').value = params.get('utm_campaign') || '';

      form.addEventListener('submit', async (event) => {
        event.preventDefault();
        formAlert.classList.remove('show');
        const consent = document.querySelector('#contactConsent');
        if (!consent.checked) {
          document.querySelector('#consent-error').textContent = 'Please confirm how we may contact you about this application.';
          consent.focus();
          return;
        }
        document.querySelector('#consent-error').textContent = '';

        submitButton.disabled = true;
        submitButton.textContent = 'Sending...';
        const data = Object.fromEntries(new FormData(form).entries());

        try {
          if (data.website) return;
          const endpoint = document.querySelector('meta[name="twins-employment-endpoint"]').content.trim();
          let reference = `APP-DEMO-${Date.now().toString().slice(-6)}`;

          if (endpoint) {
            const response = await fetch(endpoint, {
              method: 'POST',
              credentials: 'same-origin',
              headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
              body: JSON.stringify(data)
            });
            const result = await response.json().catch(() => ({}));
            if (!response.ok) throw new Error(result?.error?.message || 'We could not send your application. Please try again.');
            reference = result.applicationId || reference;
          } else {
            await new Promise((resolve) => setTimeout(resolve, 700));
          }

          form.style.display = 'none';
          document.querySelector('#reference').textContent = `Reference: ${reference}`;
          successPanel.classList.add('show');
          successPanel.scrollIntoView({ behavior: 'smooth', block: 'center' });
        } catch (error) {
          formAlert.textContent = error.message || 'We could not send your application. Please try again.';
          formAlert.classList.add('show');
        } finally {
          submitButton.disabled = false;
          submitButton.textContent = 'Send my application';
        }
      });
    })();
  </script>
